#include <tunables/global>

# Custom AppArmor profile for the Renault 5 add-on, based on Home Assistant's recommended
# add-on profile (https://developers.home-assistant.io/docs/add-ons/presentation).
#
# The s6-overlay v3 boot chain execs a long chain of shell + execline scripts (/init →
# preinit → stage0 → cont-init.d → services.d → bashio /run.sh), and each interpreter must
# be able to *read* the next script. An earlier release tried to enumerate read-only system
# paths and granted only `ix` (execute) on those trees — so the very first step,
# `/bin/sh /init`, was denied ("can't open '/init': Permission denied") and the add-on
# could not start after updating. AppArmor is only enforced by the Supervisor (not by a
# local `docker run`, and not in CI), so this could not be caught before release.
#
# This profile therefore follows HA's reference: broad `file`/`capability` access so the
# boot chain and bashio always start, with the security value concentrated in the explicit
# network limits and the denied escalation primitives below. Shipping any custom profile
# still earns the Supervisor's +1 "custom AppArmor" rating point.
profile renault_5 flags=(attach_disconnected,mediate_deleted) {
  #include <abstractions/base>

  # Let the s6-overlay supervision tree + bashio entrypoint read/exec their own scripts and
  # use the capabilities/signals they need (suexec, fix-attrs, service signalling). The add-on
  # runs as root, so withholding these only risks breaking the boot — not meaningful defence.
  file,
  capability,
  signal,

  # Networking the poller actually needs: outbound TLS to Kamereon + MQTT (tcp), DNS (udp),
  # interface lookups over netlink, and the s6 tree's AF_UNIX sockets. No AF_PACKET (raw)
  # socket is granted, so a compromised dependency cannot sniff or spoof on the wire.
  network inet tcp,
  network inet6 tcp,
  network inet udp,
  network inet6 udp,
  network netlink raw,
  unix,

  # Deny the escalation primitives the add-on never uses, even though it runs as root.
  deny mount,
  deny umount,
  deny pivot_root,
  deny ptrace,

  /dev/tty rw,
}
